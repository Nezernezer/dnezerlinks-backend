<script>
// This script connects your HTML page to your Render backend
const userEmail = localStorage.getItem('userEmail'); 

async function fetchMyVirtualAccount() {
    try {
        const response = await fetch('https://dnezerlinks-backend.onrender.com/get-virtual-account', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ email: userEmail })
        });

        const data = await response.json();

        if (data.account_number) {
            // These IDs must match the ones in your HTML (bankName, accNumber, accName)
            document.getElementById('bankName').innerText = data.bank_name;
            document.getElementById('accNumber').innerText = data.account_number;
            document.getElementById('accName').innerText = data.account_name;
        } else {
            console.error("Account error:", data.error);
        }
    } catch (error) {
        console.error("Connection Error:", error);
    }
}

if (userEmail) {
    fetchMyVirtualAccount();
} else {
    console.log("No user email found in localStorage. Redirecting...");
    // window.location.href = 'login.html'; 
}
</script>
